  <script>
    // 🔑 CONFIGURA TU FIREBASE AQUÍ
    const firebaseConfig = {
      apiKey: "TU_API_KEY",
      authDomain: "TU_PROYECTO.firebaseapp.com",
      projectId: "TU_PROYECTO",
      storageBucket: "TU_PROYECTO.appspot.com",
      messagingSenderId: "TU_ID",
      appId: "TU_APP_ID"
    };

    firebase.initializeApp(firebaseConfig);
    const db = firebase.firestore();

    // 🌐 ESTADO GLOBAL
    let currentRole = null, currentDriver = null, activeRide = null;
    let psgLat = null, psgLon = null, drvLat = null, drvLon = null;
    const MAX_KM = 3;
    let unsubRides = null, unsubLost = null, unsubDrivers = null;

    // 📐 HAVERSINE
    function calcDistance(lat1, lon1, lat2, lon2) {
      const R = 6371;
      const dLat = (lat2 - lat1) * Math.PI / 180;
      const dLon = (lon2 - lon1) * Math.PI / 180;
      const a = Math.sin(dLat/2)**2 + Math.cos(lat1*Math.PI/180)*Math.cos(lat2*Math.PI/180)*Math.sin(dLon/2)**2;
      return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
    }

    // 🧹 LIMPIAR LISTENERS
    function cleanupListeners() {
      unsubRides?.(); unsubLost?.(); unsubDrivers?.();
      unsubRides = unsubLost = unsubDrivers = null;
    }

    // 📍 GPS
    function getPassengerLocation() {
      if(!navigator.geolocation) return alert('🌍 GPS no disponible');
      document.getElementById('btn-get-loc').disabled = true;
      document.getElementById('psg-loc-status').textContent = '📡 Obteniendo...';
      navigator.geolocation.getCurrentPosition(pos => {
        psgLat = pos.coords.latitude; psgLon = pos.coords.longitude;
        document.getElementById('psg-loc-status').textContent = `✅ ${psgLat.toFixed(4)}, ${psgLon.toFixed(4)}`;
        document.getElementById('btn-request').disabled = false;
      }, err => {
        document.getElementById('psg-loc-status').textContent = '❌ Activa GPS y permisos';
        document.getElementById('btn-get-loc').disabled = false;
      }, {enableHighAccuracy: true, timeout: 10000});
    }

    function getDriverLocation() {
      if(!navigator.geolocation) return alert('🌍 GPS no disponible');
      document.getElementById('drv-loc-status').textContent = '📡 Actualizando...';
      navigator.geolocation.getCurrentPosition(pos => {
        drvLat = pos.coords.latitude; drvLon = pos.coords.longitude;
        document.getElementById('drv-loc-status').textContent = `📍 ${drvLat.toFixed(4)}, ${drvLon.toFixed(4)}`;
        if(currentDriver) {
          db.collection('drivers').doc(currentDriver.id).update({ lat: drvLat, lon: drvLon })
            .catch(err => console.warn('Error actualizando ubicación:', err));
        }
      }, err => {
        document.getElementById('drv-loc-status').textContent = '❌ Error GPS';
      }, {enableHighAccuracy: true});
    }

    // 🧭 NAVEGACIÓN
    function setRole(role) {
      cleanupListeners();
      currentRole = role;
      document.querySelectorAll('[id^="panel-"]').forEach(p => p.classList.add('hidden'));
      document.getElementById('bottom-nav').classList.toggle('hidden', !role);
      document.querySelectorAll('.nav button').forEach(b => b.classList.remove('active'));
      
      if(role === 'admin') {
        document.getElementById('panel-admin').classList.remove('hidden');
        loadAdminDrivers();
      } else if(role === 'driver') {
        document.getElementById('panel-driver').classList.remove('hidden');
      } else if(role === 'passenger') {
        document.getElementById('panel-passenger').classList.remove('hidden');
        loadLostItems();
      } else {
        document.getElementById('panel-role').classList.remove('hidden');
      }
      if(role) document.getElementById(`nav-${role==='passenger'?'psg':role}`).classList.add('active');
    }

    // 👨‍💼 ADMIN
    async function saveDriver() {
      const id = document.getElementById('adm-id').value || 'DRV-'+Date.now().toString().slice(-4);
      const file = document.getElementById('adm-photo').files[0];
      const reader = new FileReader();
      reader.onload = async e => {
        const data = {
          id, name: document.getElementById('adm-name').value || 'Conductor',
          site: document.getElementById('adm-site').value,
          vehicle: document.getElementById('adm-vehicle').value,
          color: document.getElementById('adm-color').value,
          active: false, photo: e.target.result || null,
          qrToken: 'QR-'+Math.random().toString(36).substr(2,6).toUpperCase(),
          createdAt: firebase.firestore.FieldValue.serverTimestamp()
        };
        try {
          await db.collection('drivers').doc(id).set(data);
          alert('✅ Conductor registrado en la nube');
          document.querySelectorAll('#panel-admin input').forEach(i => i.value = '');
        } catch(err) {
          alert('❌ Error: ' + err.message);
        }
      };
      file ? reader.readAsDataURL(file) : reader.onload({target:{result:null}});
    }

    function loadAdminDrivers() {
      const list = document.getElementById('adm-list');
      unsubDrivers = db.collection('drivers').orderBy('createdAt', 'desc').onSnapshot(snap => {
        list.innerHTML = snap.docs.map(doc => {
          const d = doc.data();
          return `
          <div class="card">
            <div style="display:flex;justify-content:space-between;align-items:center;">
              <div><strong>${d.id}</strong> • ${d.name}<br><small>${d.site} • ${d.color} • ${d.vehicle}</small></div>
              <span class="chip" style="background:${d.active?'#dcfce7':'#fee2e2'}">${d.active?'Activo':'Inactivo'}</span>
            </div>
            <div class="qr-box" style="margin-top:8px;"><small>CÓDIGO QR</small><br><code>${d.qrToken}</code></div>
            <button class="btn btn-danger" style="margin-top:8px;padding:8px;" onclick="deleteDriver('${d.id}')">🗑️ Eliminar</button>
          </div>`;
        }).join('');
      });
    }

    async function deleteDriver(id) {
      if(!confirm('¿Eliminar conductor?')) return;
      await db.collection('drivers').doc(id).delete();
    }

    // 🏍️ DRIVER
    async function loginDriver() {
      const id = document.getElementById('drv-id-input').value.trim();
      try {
        const doc = await db.collection('drivers').doc(id).get();
        if(!doc.exists) return alert('❌ ID no encontrado');
        currentDriver = { id, ...doc.data() };
        document.getElementById('drv-login').classList.add('hidden');
        document.getElementById('drv-dashboard').classList.remove('hidden');
        document.getElementById('drv-name').textContent = currentDriver.name;
        document.getElementById('drv-meta').textContent = `${currentDriver.vehicle} • ${currentDriver.color} • ${currentDriver.site}`;
        document.getElementById('drv-avatar').style.backgroundImage = currentDriver.photo ? `url(${currentDriver.photo})` : '';
        document.getElementById('drv-avatar').textContent = currentDriver.photo ? '' : '👤';
        updateDriverStatusUI();
        loadDriverRides();
      } catch(err) {
        alert('❌ Error de conexión: ' + err.message);
      }
    }

    async function toggleDriverStatus() {
      if(!currentDriver) return;
      const newStatus = !currentDriver.active;
      await db.collection('drivers').doc(currentDriver.id).update({ active: newStatus });
      currentDriver.active = newStatus;
      updateDriverStatusUI();
    }

    function updateDriverStatusUI() {
      const btn = document.getElementById('btn-toggle-status');
      btn.textContent = currentDriver.active ? '🔴 Desactivar' : '✅ Activar';
      btn.className = `btn ${currentDriver.active ? 'btn-danger' : 'btn-success'}`;
    }

    function loadDriverRides() {
      unsubRides = db.collection('rides').where('status', '==', 'PENDING').onSnapshot(snap => {
        if(!drvLat || !drvLon) {
          document.getElementById('drv-rides').innerHTML = '<p class="loading">📍 Activa tu GPS para ver solicitudes</p>';
          return;
        }
        const nearby = snap.docs.map(d => ({id: d.id, ...d.data()}))
          .filter(r => r.lat && r.lon)
          .map(r => ({...r, dist: calcDistance(r.lat, r.lon, drvLat, drvLon)}))
          .filter(r => r.dist <= MAX_KM)
          .sort((a,b) => a.dist - b.dist);
          
        document.getElementById('drv-rides').innerHTML = nearby.length ? nearby.map(r => `
          <div class="ride-card">
            <div class="info"><strong>Pasajero: ${r.passengerName}</strong><p>📍 A ${r.dist.toFixed(1)} km • Tarifa: S/. ${r.fare || '?'}</p></div>
            <button class="btn btn-success" style="width:auto;padding:8px 12px;" onclick="acceptRide('${r.id}')">Aceptar</button>
          </div>
        `).join('') : '<p class="loading">📡 Sin solicitudes en '+MAX_KM+' km</p>';
      });
    }

    async function acceptRide(rideId) {
      if(!confirm('¿Aceptar este viaje?')) return;
      await db.collection('rides').doc(rideId).update({
        status: 'ACCEPTED',
        driverId: currentDriver.id,
        acceptedAt: firebase.firestore.FieldValue.serverTimestamp()
      });
      alert('✅ Viaje aceptado. Espera confirmación del pasajero.');
    }

    function logoutDriver() {
      cleanupListeners();
      currentDriver = null; drvLat = null; drvLon = null;
      document.getElementById('drv-dashboard').classList.add('hidden');
      document.getElementById('drv-login').classList.remove('hidden');
      document.getElementById('drv-id-input').value = '';
    }

    // 🙋 PASSENGER
    async function requestRide() {
      const name = document.getElementById('psg-name').value.trim();
      if(!name) return alert('Ingresa tu nombre');
      if(!psgLat) return alert('Activa tu ubicación GPS primero');
      
      const rideRef = db.collection('rides');
      const newRide = await rideRef.add({
        passengerName: name,
        lat: psgLat, lon: psgLon,
        status: 'PENDING', driverId: null,
        fare: 5 + Math.floor(Math.random()*5),
        createdAt: firebase.firestore.FieldValue.serverTimestamp()
      });
      activeRide = { id: newRide.id, ...await newRide.get().data() };
      
      document.getElementById('psg-ride-status').classList.remove('hidden');
      document.getElementById('psg-status-text').textContent = '🔍 Busciendo conductores...';
      document.getElementById('psg-negotiation').classList.remove('hidden');
      document.getElementById('psg-fare').value = activeRide.fare;
      watchMyRide();
    }

    function watchMyRide() {
      if(!activeRide) return;
      unsubRides = db.collection('rides').doc(activeRide.id).onSnapshot(doc => {
        if(!doc.exists) return;
        const ride = doc.data();
        activeRide = { id: doc.id, ...ride };
        
        if(ride.status === 'ACCEPTED') {
          document.getElementById('psg-status-text').textContent = '✅ Conductor en camino';
          document.getElementById('psg-status-text').className = 'status-box status-active';
          document.getElementById('psg-negotiation').classList.add('hidden');
          document.getElementById('psg-active-ride').classList.remove('hidden');
          
          db.collection('drivers').doc(ride.driverId).get().then(d => {
            const drv = d.data();
            document.getElementById('psg-driver-info').textContent = `👤 ${drv.name} • 🏍️ ${drv.vehicle} (${drv.color})`;
            const dist = calcDistance(ride.lat, ride.lon, drv.lat||ride.lat, drv.lon||ride.lon);
            document.getElementById('psg-driver-dist').textContent = `📍 Distancia inicial: ~${dist.toFixed(1)} km`;
          });
        }
      });
    }

    async function acceptFare() {
      if(!activeRide) return;
      const fare = parseFloat(document.getElementById('psg-fare').value) || activeRide.fare;
      await db.collection('rides').doc(activeRide.id).update({ fare });
      alert('💬 Tarifa enviada. Espera confirmación...');
    }

    async function cancelRide() {
      if(!activeRide) return;
      await db.collection('rides').doc(activeRide.id).update({ status: 'CANCELLED' });
      document.getElementById('psg-ride-status').classList.add('hidden');
      activeRide = null;
    }

    async function completeRide() {
      if(!activeRide) return;
      await db.collection('rides').doc(activeRide.id).update({ status: 'COMPLETED', completedAt: firebase.firestore.FieldValue.serverTimestamp() });
      alert('💵 Pago registrado. ¡Gracias!');
      document.getElementById('psg-ride-status').classList.add('hidden');
      activeRide = null;
    }

    async function reportLost() {
      const desc = document.getElementById('lost-desc').value.trim();
      const contact = document.getElementById('lost-contact').value.trim();
      if(!desc) return alert('Describe el objeto');
      
      await db.collection('lostItems').add({
        description: desc, contact, status: 'REPORTADO',
        date: new Date().toLocaleString(),
        reportedAt: firebase.firestore.FieldValue.serverTimestamp()
      });
      document.getElementById('lost-desc').value = '';
      alert('📢 Reporte enviado');
    }

    function loadLostItems() {
      unsubLost = db.collection('lostItems').orderBy('reportedAt', 'desc').limit(10).onSnapshot(snap => {
        document.getElementById('lost-list').innerHTML = snap.docs.map(d => {
          const i = d.data();
          return `<div class="card" style="padding:10px;"><strong>📦 ${i.description}</strong><br><small>📞 ${i.contact||'N/A'} • ${i.status} • ${i.date}</small></div>`;
        }).join('');
      });
    }

    // 📲 SERVICE WORKER INLINE (PWA básico)
    if('serviceWorker' in navigator) {
      const swCode = `self.addEventListener('install', e => self.skipWaiting()); self.addEventListener('fetch', e => e.respondWith(fetch(e.request).catch(() => new Response('Offline'))));`;
      navigator.serviceWorker.register(URL.createObjectURL(new Blob([swCode], {type:'application/javascript'}))).catch(()=>{});
    }
